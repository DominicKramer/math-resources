We'll expand about the first row since there are no zeros to exploit,
%
\begin{align*}
\begin{vmatrix}
 -2 & 3 & -2 \\
 -4 & -2 & 1 \\
 2 & 4 & 2
\end{vmatrix}
&=
(-2)\begin{vmatrix}
 -2 & 1 \\
 4 & 2
\end{vmatrix}
+
(-1)(3)\begin{vmatrix}
 -4 & 1 \\
 2 & 2
\end{vmatrix}
+
(-2)\begin{vmatrix}
 -4 & -2 \\
 2 & 4
\end{vmatrix}\\
%
&=(-2)((-2)(2)-1(4)) + (-3)((-4)(2)-1(2)) + (-2)((-4)(4)-(-2)(2))\\
&=(-2)(-8)+(-3)(-10) + (-2)(-12)=70
\end{align*}
