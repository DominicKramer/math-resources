\begin{enumerate}
%
\item Construct the elementary matrix that will effect the row operation $\rowopadd{-6}{2}{3}$ on a $4\times 7$ matrix.
%
\item Compute the determinant of the matrix
%
\begin{equation*}
\begin{bmatrix}
2&3&-1\\
3&8&2\\
4&-1&-3
\end{bmatrix}
\end{equation*}
%
\item Compute the determinant of the matrix
%
\begin{equation*}
\begin{bmatrix}
3 & 9 & -2 & 4 & 2 \\
0 & 1 & 4 & -2 & 7 \\
0 & 0 & -2 & 5 & 2 \\
0 & 0 & 0  & -1 & 6 \\
0 & 0 & 0  & 0 & 4 
\end{bmatrix}
\end{equation*}
%
\end{enumerate}
