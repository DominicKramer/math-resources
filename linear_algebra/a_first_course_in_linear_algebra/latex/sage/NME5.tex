We can easily illustrate our latest equivalence for nonsingular matrices.
%
%
\begin{sageexample}
sage: A = matrix(QQ, [[ 2,  3, -3,  2,  8, -4],
...                   [ 3,  4, -4,  4,  8,  1],
...                   [-2, -2,  3, -3, -2, -7],
...                   [ 0,  1, -1,  2,  3,  4],
...                   [ 2,  1,  0,  1, -4,  4],
...                   [ 1,  2, -2,  1,  7, -5]])
sage: not A.is_singular()
True
sage: V = QQ^6
sage: cols = A.columns()
sage: V == V.span(cols)
True
sage: V.linear_dependence(cols) == []
True
\end{sageexample}
